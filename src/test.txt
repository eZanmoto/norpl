fn new_set() {
    return {
        'values': {},
        'set': fn (k) {
            this[k] = true;
        },
        'has': fn (k) {
            return this[k];
        },
    };
}

let xs = new_set();

xs.set('a');
xs.set('b');
print(xs.has('a'));

xs['set']('a');
xs['set']('b');
print(xs['has']('a'));

xs = {};

xs.abc = 1;
xs.abc = 1 + xs.abc;
xs.abc = xs.abc + 1;
print(xs.abc);

print(xs.this);
